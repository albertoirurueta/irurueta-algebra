31  class MultivariateGaussianRandomizerTest {
32  
33      private static final int NUM_SAMPLES = 1000000;
34      private static final double RELATIVE_ERROR = 0.05;
35  
36      private static final double MIN_RANDOM_VALUE = -100.0;
37      private static final double MAX_RANDOM_VALUE = 100.0;
38  
39      @Test
40      void testConstructor() throws WrongSizeException, InvalidCovarianceMatrixException {
41          // test empty constructor
42          var randomizer = new MultivariateGaussianRandomizer();
43  
44          // check correctness
45          assertArrayEquals(new double[]{0.0}, randomizer.getMean(), 0.0);
46          assertEquals(Matrix.identity(1, 1), randomizer.getCovariance());
47  
48  
49          // test constructor with random
50          randomizer = new MultivariateGaussianRandomizer(new Random());
51  
52          // check correctness
53          assertArrayEquals(new double[]{0.0}, randomizer.getMean(), 0.0);
54          assertEquals(Matrix.identity(1, 1), randomizer.getCovariance());
55  
56          // Force NullPointerException
57          assertThrows(NullPointerException.class, () -> new MultivariateGaussianRandomizer(null));
58  
59          // test constructor with mean and covariance
60          final var uniRandomizer = new UniformRandomizer(new Random());
61          final var mean = new double[2];
62          uniRandomizer.fill(mean, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
63          final var cov = DecomposerHelper.getSymmetricPositiveDefiniteMatrixInstance(
64                  DecomposerHelper.getLeftLowerTriangulatorFactor(2));
65  
66          randomizer = new MultivariateGaussianRandomizer(mean, cov);
67  
68          // check correctness
69          assertArrayEquals(mean, randomizer.getMean(), 0.0);
70          assertEquals(cov, randomizer.getCovariance());
71  
72          // Force WrongSizeException
73          assertThrows(WrongSizeException.class, () -> new MultivariateGaussianRandomizer(new double[1], cov));
74  
75          // Force InvalidCovarianceMatrixException
76          assertThrows(InvalidCovarianceMatrixException.class, () -> new MultivariateGaussianRandomizer(mean,
77                  new Matrix(2, 2)));
78  
79          // test constructor with random, mean and covariance
80          randomizer = new MultivariateGaussianRandomizer(new Random(), mean, cov);
81  
82          // check correctness
83          assertArrayEquals(mean, randomizer.getMean(), 0.0);
84          assertEquals(cov, randomizer.getCovariance());
85  
86          // Force WrongSizeException
87          assertThrows(WrongSizeException.class, () -> new MultivariateGaussianRandomizer(new Random(), new double[1],
88                  cov));
89  
90          // Force InvalidCovarianceMatrixException
91          assertThrows(InvalidCovarianceMatrixException.class, () -> new MultivariateGaussianRandomizer(new Random(),
92                  mean, new Matrix(2, 2)));
93      }

95      @Test
96      void testGetSetMeanAndCovariance() throws WrongSizeException, InvalidCovarianceMatrixException {
97          final var randomizer = new MultivariateGaussianRandomizer();
98  
99          // check initial values
100         assertArrayEquals(new double[]{0.0}, randomizer.getMean(), 0.0);
101         assertEquals(Matrix.identity(1, 1), randomizer.getCovariance());
102 
103         // set new values
104         final var uniRandomizer = new UniformRandomizer(new Random());
105         final var mean = new double[2];
106         uniRandomizer.fill(mean, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
107         final var cov = DecomposerHelper.getSymmetricPositiveDefiniteMatrixInstance(
108                 DecomposerHelper.getLeftLowerTriangulatorFactor(2));
109 
110         randomizer.setMeanAndCovariance(mean, cov);
111 
112         // check correctness
113         assertArrayEquals(mean, randomizer.getMean(), 0.0);
114         assertEquals(cov, randomizer.getCovariance());
115 
116         // Force WrongSizeException
117         assertThrows(WrongSizeException.class, () -> randomizer.setMeanAndCovariance(new double[1], cov));
118 
119         // Force InvalidCovarianceMatrixException
120         assertThrows(InvalidCovarianceMatrixException.class, () -> randomizer.setMeanAndCovariance(mean,
121                 new Matrix(2, 2)));
122     }

124     @Test
125     void testNext() throws WrongSizeException, InvalidCovarianceMatrixException, DecomposerException {
126 
127         final var uniRandomizer = new UniformRandomizer(new Random());
128         final var mean = new double[2];
129         uniRandomizer.fill(mean, MIN_RANDOM_VALUE, MAX_RANDOM_VALUE);
130         final var cov = DecomposerHelper.getSymmetricPositiveDefiniteMatrixInstance(
131                 DecomposerHelper.getLeftLowerTriangulatorFactor(2));
132 
133         final var randomizer = new MultivariateGaussianRandomizer(mean, cov);
134 
135         final var mean1 = new double[2];
136         final var mean2 = new double[2];
137 
138         final var values1 = new double[2];
139         double[] values2;
140         double[] tmp1;
141         double[] tmp2;
142 
143         final var col1 = new Matrix(2, 1);
144         final var row1 = new Matrix(1, 2);
145         final var sqr1 = new Matrix(2, 2);
146 
147         final var col2 = new Matrix(2, 1);
148         final var row2 = new Matrix(1, 2);
149         final var sqr2 = new Matrix(2, 2);
150 
151         final var sqrSum1 = new Matrix(2, 2);
152         final var sqrSum2 = new Matrix(2, 2);
153         for (var i = 0; i < NUM_SAMPLES; i++) {
154             randomizer.next(values1);
155             values2 = randomizer.next();
156 
157             tmp1 = Arrays.copyOf(values1, 2);
158             tmp2 = Arrays.copyOf(values2, 2);
159 
160             ArrayUtils.multiplyByScalar(tmp1, 1.0 / (double) NUM_SAMPLES, tmp1);
161             ArrayUtils.multiplyByScalar(tmp2, 1.0 / (double) NUM_SAMPLES, tmp2);
162 
163             ArrayUtils.sum(mean1, tmp1, mean1);
164             ArrayUtils.sum(mean2, tmp2, mean2);
165 
166             col1.fromArray(values1);
167             row1.fromArray(values1);
168             col1.multiply(row1, sqr1);
169             sqr1.multiplyByScalar(1.0 / (double) NUM_SAMPLES);
170 
171             col2.fromArray(values2);
172             row2.fromArray(values2);
173             col2.multiply(row2, sqr2);
174             sqr2.multiplyByScalar(1.0 / (double) NUM_SAMPLES);
175 
176             sqrSum1.add(sqr1);
177             sqrSum2.add(sqr2);
178         }
179 
180         col1.fromArray(mean1);
181         row1.fromArray(mean1);
182         var sqrMean1 = col1.multiplyAndReturnNew(row1);
183 
184         col2.fromArray(mean2);
185         row2.fromArray(mean2);
186         var sqrMean2 = col2.multiplyAndReturnNew(row2);
187 
188         final var cov1 = sqrSum1.subtractAndReturnNew(sqrMean1);
189         final var cov2 = sqrSum2.subtractAndReturnNew(sqrMean2);
190 
191         // check correctness
192         final var maxMean = Math.max(ArrayUtils.max(mean), Math.abs(ArrayUtils.min(mean)));
193         assertArrayEquals(mean1, mean, RELATIVE_ERROR * maxMean);
194         assertArrayEquals(mean2, mean, RELATIVE_ERROR * maxMean);
195 
196         final var detCov = Utils.det(cov);
197         assertTrue(cov1.equals(cov, RELATIVE_ERROR * detCov));
198         assertTrue(cov2.equals(cov, RELATIVE_ERROR * detCov));
199     }
